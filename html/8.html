body {
    /*tarayici elemanlarinin margin degerini 0 yapiyor*/
    margin: 0;
    /*body ile tarayicinin ust kismi arasinda 220px bosluk kalacak. buraya header yerlesecek*/
    padding-top: 220px;
}

header {
    /*header elemaninin arkaplan rengi*/
    background-color: aqua;
    /*header elemanininn uzunluk degeri body ile tarayicinin ust kismi arasindaki bosluk kadar*/
    height: 220px;
    /*header elemaninin eni tarayiciyi kaplasin*/
    width: 100%;
    /*header elemaninin tarayicinin en ust noktasinda surekli kalabilmesini saglar*/
    position: fixed;
    top: 0px;
}

.yapiskan_menu {
    /*nav elemanının header elemaninin ustunde kalabilmesini saglar*/
    position: relative;
    /*nav elemanının eninin tarayıcı eni kadar olmasını sağlar*/
    width: 100%;
    /*nav elemanının arka plan rengi*/
    background-color: #330303;
    /*nav elemaninin yukseklik degeri*/
    height: 50px;
    /*nav elemani tarayicinin ust noktasina geldigi zaman position:fixed oluyor. bu durumda body elemaninin konumunun degismemesi icin height kadar margin-bottom verilir*/
    margin-bottom: -50px;
    /*menunun yazi body elemaninin uzerinde kalabilmesi icin body elemaninin z-index inden buyuk olmali*/
    z-index: 10;
}

    /*linkteki (a) yazinin rengini değiştirir*/

    /*linkin (a) altındaki çizğiyi kaldırır*/
    text-decoration: none;
    /*linkin (a) paragraph gibi gorunmesını saglar*/
    display: block;
    /*linklerin (a) sagdan sola sıralanmasını sağlar*/
    float: left;
}

style.css 

var ym = $("#yapiskanMenu");
var ymt = "yapiskan_menu_tepede";
var hdr = $('header').height();

$(window).scroll(function () {
    if ($(this).scrollTop() > hdr) /*scrollbar headerin height degeri kadar kaydirildi*/ {
        ym.addClass(ymt); //yapiskan menu ye yeni css ekleniyor
    } else {
        ym.removeClass(ymt);
    }
});
